## 描述
给出一个含有括号的字符串，判断其是否正确的闭合例如，'[](){}' 是正确闭合的，'{[()]}' 也是正确闭合的，但 '[](}' 这样的就不是
## 示例
Input: "()"
Output: true

Input: "()[]{}"
Output: true

Input: "(]"
Output: true
## 代码
```js
  var isValid = function(s) {
    let validate = true;
    const stack = [];
    const map = {
      ')':'(',
      ']':'[',
      '}':'{'
    };

    for(var i = 0; i < s.length; i += 1) {
      var str = s[i];
      if (map[str]) {
        if (stack.pop() !== map(str)) {
          return false;
        }
      } else {
        stack.push(str);
      }
    }
    return stack.length === 0;
  }
```
